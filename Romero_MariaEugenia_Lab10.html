wcss = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters = i, random_state = 42)
    kmeans.fit(X)    
    wcss.append(kmeans.inertia_)
sns.lineplot(x=range(1, 11), y=wcss,marker='X',color='blue')
plt.title('Elbow Method')
plt.xlabel('Number of clusters')
plt.ylabel('WCSS')
plt.show()
